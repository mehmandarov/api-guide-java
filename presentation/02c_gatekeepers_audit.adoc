link:../src/main/java/com/mehmandarov/confapi/gatekeepers/AuditFilter.java[AuditFilter.java]

[source,java]
----
@Provider
@Audited                        // ← Only runs on @Audited endpoints
public class AuditFilter
        implements ContainerRequestFilter, ContainerResponseFilter {

    @Override // ← BEFORE the resource method — log the incoming request
    public void filter(ContainerRequestContext request) {
        request.setProperty(START_TIME, System.nanoTime());
        LOG.info(() -> String.format("[AUDIT] >>> %s %s | caller=%s",
                request.getMethod(),
                request.getUriInfo().getRequestUri(),
                principalName(request.getSecurityContext())));
    }

    @Override // ← AFTER the resource method — log status + elapsed time
    public void filter(ContainerRequestContext request,
                       ContainerResponseContext response) {
        long elapsedMs = (System.nanoTime() - startNanos) / 1_000_000;
        LOG.info(() -> String.format("[AUDIT] <<< %s | status=%d | elapsed=%dms",
                request.getMethod(), response.getStatus(), elapsedMs));
    }
}
----
